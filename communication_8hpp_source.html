131 recv(mpi::communicator& comm, const mpi::status& status)
132 {
133  if (info.from == -1) // uninitialized
134  {
135  MemoryBuffer bb;
136  comm.recv(status.source(), status.tag(), bb.buffer);
137 
138  if (status.tag() == tags::piece) // first piece is the header
139  {
140  size_t msg_size;
141  diy::load(bb, msg_size);
142  diy::load(bb, info);
143 
144  message.buffer.reserve(msg_size);
145  }
146  else // tags::queue
147  {
148  diy::load_back(bb, info);
149  message.swap(bb);
150  }
151  }
152  else
153  {
154  size_t start_idx = message.buffer.size();
155  size_t count = status.count<char>();
156  message.buffer.resize(start_idx + count);
157 
158  detail::VectorWindow<char> window;
159  window.begin = &message.buffer[start_idx];
160  window.count = count;
161 
162  comm.recv(status.source(), status.tag(), window);
163  }
164 
165  if (status.tag() == tags::queue)
166  done = true;
167 }
