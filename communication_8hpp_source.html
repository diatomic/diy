97  if (info.from == -1) // uninitialized
98  {
99  MemoryBuffer bb;

101 
102  diy::load_back(bb, info);
103  info.nparts--;
104  if (info.nparts > 0) // multi-part message
105  {
106  size_t msg_size;
107  diy::load(bb, msg_size);
108  message.buffer.reserve(msg_size);
109  } else
110  message.swap(bb);
111 
112  result = true;
113  }
114  else
115  {
116  size_t start_idx = message.buffer.size();
117  size_t count = status.count<char>();
118  message.buffer.resize(start_idx + count);
119 
120  detail::VectorWindow<char> window;
121  window.begin = &message.buffer[start_idx];
122  window.count = count;
123 
124  comm.recv(status.source(), status.tag(), window);
125 
126  info.nparts--;
127  }
128 
129  if (info.nparts == 0)
130  done = true;
131

133 }
