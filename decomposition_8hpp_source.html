295 diy::RegularDecomposer<Bounds>::
296 decompose(int rank, const StaticAssigner& assigner, const Creator& create)
297 {
298  std::vector<int> gids;
299  assigner.local_gids(rank, gids);
300  for (int i = 0; i < (int)gids.size(); ++i)
301  {
302  int gid = gids[i];
303 
304  DivisionsVector coords;
305  gid_to_coords(gid, coords);
306 
307  Bounds core(dim), bounds(dim);
308  fill_bounds(core, coords);
309  fill_bounds(bounds, coords, true);
310 
311  // Fill link with all the neighbors
312  Link link(dim, core, bounds);
313  std::vector<int> offsets(dim, -1);
314  offsets[0] = -2;
315  while (!all(offsets, 1))
316  {
317  // next offset
318  int j;
319  for (j = 0; j < dim; ++j)
320  if (offsets[j] == 1)
321  offsets[j] = -1;
322  else
323  break;
324  ++offsets[j];
325 
326  if (all(offsets, 0)) continue; // skip ourselves
327 
328  DivisionsVector nhbr_coords(dim);
329  Direction dir(dim,0), wrap_dir(dim,0);
330  bool inbounds = true;
331  for (int k = 0; k < dim; ++k)
332  {
333  nhbr_coords[k] = coords[k] + offsets[k];
334 
335  // wrap
336  if (nhbr_coords[k] < 0)
337  {
338  if (wrap[k])
339  {
340  nhbr_coords[k] = divisions[k] - 1;
341  wrap_dir[k] = -1;
342  }
343  else
344  inbounds = false;
345  }
346 
347  if (nhbr_coords[k] >= divisions[k])
348  {
349  if (wrap[k])
350  {
351  nhbr_coords[k] = 0;
352  wrap_dir[k] = 1;
353  }
354  else
355  inbounds = false;
356  }
357 
358  // NB: this needs to match the addressing scheme in dir_t (in constants.h)
359  if (offsets[k] == -1 || offsets[k] == 1)
360  dir[k] = offsets[k];
361  }
362  if (!inbounds) continue;
363 
364  int nhbr_gid = coords_to_gid(nhbr_coords);
365  BlockID bid; bid.gid = nhbr_gid; bid.proc = assigner.rank(nhbr_gid);
366  link.add_neighbor(bid);
367 
368  Bounds nhbr_core(dim);
369  fill_bounds(nhbr_core, nhbr_coords);
370  link.add_core(nhbr_core);
371 
372  Bounds nhbr_bounds(dim);
373  fill_bounds(nhbr_bounds, nhbr_coords, true);
374  link.add_bounds(nhbr_bounds);
375

377  link.add_wrap(wrap_dir);
378  }

514 template<class Bounds>
515 void
516 diy::RegularDecomposer<Bounds>::
517 fill_divisions(std::vector<int>& divisions_) const
518 {
519  // prod = number of blocks unconstrained by user; c = number of unconstrained dimensions
520  int prod = 1; int c = 0;
521  for (int i = 0; i < dim; ++i)
522  if (divisions_[i] != 0)
523  {
524  prod *= divisions_[i];
525  ++c;
526  }
527 
528  if (nblocks % prod != 0)
529  throw std::runtime_error("Total number of blocks cannot be factored into provided divs");
530 
531  if (c == (int) divisions_.size()) // nothing to do; user provided all divs
532  return;
533 
534  // factor number of blocks left in unconstrained dimensions
535  // factorization is sorted from smallest to largest factors
536  std::vector<unsigned> factors;
537  factor(factors, nblocks/prod);
538 
539  using detail::Div;
540  std::vector< Div<Coordinate> > missing_divs; // pairs consisting of (dim, #divs)
541 
542  // init missing_divs
543  for (int i = 0; i < dim; i++)
544  {
545  if (divisions_[i] == 0)
546  {
547  Div<Coordinate> div;
548  div.dim = i;
549  div.nb = 1;
550  div.b_size = domain.max[i] - domain.min[i];
551  missing_divs.push_back(div);
552  }
553  }
554 
555  // iterate over factorization of number of blocks (factors are sorted smallest to largest)
556  for (auto f = factors.rbegin(); f != factors.rend(); ++f)
557  {
558  // fill in missing divs by dividing dimension w/ largest block size
559  // except when this would be illegal (resulting in bounds.max < bounds.min;
560  // only a problem for discrete bounds
561 
562  // sort on decreasing block size
563  std::sort(missing_divs.begin(), missing_divs.end());
564 
565  // split the dimension with the largest block size (first element in vector)
566  Coordinate min =
567  detail::BoundsHelper<Bounds>::from(0,
568  missing_divs[0].nb * (*f),
569  domain.min[missing_divs[0].dim],
570  domain.max[missing_divs[0].dim],
571  share_face[missing_divs[0].dim]);
572  Coordinate max =
573  detail::BoundsHelper<Bounds>::to(0,
574  missing_divs[0].nb * (*f),
575  domain.min[missing_divs[0].dim],
576  domain.max[missing_divs[0].dim],
577  share_face[missing_divs[0].dim]);
578  if (max >= min)
579  {
580  missing_divs[0].nb *= (*f);
581  missing_divs[0].b_size = max - min;
582  }
583  else
584  {
585  std::ostringstream oss;
586  oss << "Unable to decompose domain into " << nblocks << " blocks: " << min << " " << max;
587  throw std::runtime_error(oss.str());
588  }
589  }
590 
591  // assign the divisions
592  for (size_t i = 0; i < missing_divs.size(); i++)
593  divisions_[missing_divs[i].dim] = missing_divs[i].nb;
594 }
