185 {
186  std::vector<int> gids;
187  assigner.local_gids(rank, gids);
188  for (int i = 0; i < (int)gids.size(); ++i)
189  {
190  int gid = gids[i];
191 
192  DivisionsVector coords;
193  gid_to_coords(gid, coords);
194 
195  Bounds core, bounds;
196  fill_bounds(core, coords);
197  fill_bounds(bounds, coords, true);
198 
199  // Fill link with all the neighbors
200  Link link(dim, core, bounds);
201  std::vector<int> offsets(dim, -1);
202  offsets[0] = -2;
203  while (!all(offsets, 1))
204  {
205  // next offset
206  int i;
207  for (i = 0; i < dim; ++i)
208  if (offsets[i] == 1)
209  offsets[i] = -1;
210  else
211  break;
212  ++offsets[i];
213 
214  if (all(offsets, 0)) continue; // skip ourselves
215 
216  DivisionsVector nhbr_coords(dim);
217  int dir = 0;
218  bool inbounds = true;
219  for (int i = 0; i < dim; ++i)
220  {
221  nhbr_coords[i] = coords[i] + offsets[i];
222 
223  // wrap
224  if (nhbr_coords[i] < 0)
225  {
226  if (wrap[i])
227  {
228  nhbr_coords[i] = divisions[i] - 1;
229  link.add_wrap(Direction(1 << 2*i));
230  }
231  else
232  inbounds = false;
233  }
234 
235  if (nhbr_coords[i] >= divisions[i])
236  {
237  if (wrap[i])
238  {
239  nhbr_coords[i] = 0;
240  link.add_wrap(Direction(1 << (2*i + 1)));
241  }
242  else
243  inbounds = false;
244  }
245 
246  // NB: this needs to match the addressing scheme in dir_t (in constants.h)
247  if (offsets[i] == -1)
248  dir |= 1 << (2*i + 0);
249  if (offsets[i] == 1)
250  dir |= 1 << (2*i + 1);
251  }
252  if (!inbounds) continue;
253 
254  int nhbr_gid = coords_to_gid(nhbr_coords);
255  BlockID bid; bid.gid = nhbr_gid; bid.proc = assigner.rank(nhbr_gid);
256  link.add_neighbor(bid);
257 
258  Bounds nhbr_bounds;
259  fill_bounds(nhbr_bounds, nhbr_coords);
260  link.add_bounds(nhbr_bounds);
261 
262  link.add_direction(static_cast<Direction>(dir));
263  }

361 fill_divisions(int dim, int nblocks, std::vector<int>& divisions)
362 {
363  int prod = 1; int c = 0;
364  for (unsigned i = 0; i < dim; ++i)
365  if (divisions[i] != 0)
366  {
367  prod *= divisions[i];
368  ++c;
369  }
370 
371  if (nblocks % prod != 0)
372  {
373  std::cerr << "Incompatible requirements" << std::endl;
374  return;
375  }
376 
377  if (c == divisions.size())
378  return;
379 
380  std::vector<unsigned> factors;
381  factor(factors, nblocks/prod);
382 
383  // Fill the missing divs using LPT algorithm
384  std::vector<unsigned> missing_divs(divisions.size() - c, 1);
385  for (int i = factors.size() - 1; i >= 0; --i)
386  *std::min_element(missing_divs.begin(), missing_divs.end()) *= factors[i];
387 
388  c = 0;
389  for (unsigned i = 0; i < dim; ++i)
390  if (divisions[i] == 0)
391  divisions[i] = missing_divs[c++];
392 }
