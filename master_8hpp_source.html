410 void
411 diy::Master::
412 unload_incoming(int gid__)
413 {
414  for (IncomingRoundMap::iterator round_itr = incoming_.begin(); round_itr != incoming_.end(); ++round_itr)
415  {
416  IncomingQueuesMap::iterator qmap_itr = round_itr->second.map.find(gid__);
417  if (qmap_itr == round_itr->second.map.end())
418  {
419  continue;
420  }
421  IncomingQueuesRecords& in_qrs = qmap_itr->second;
422  for (InQueueRecords::iterator it = in_qrs.records.begin(); it != in_qrs.records.end(); ++it)
423  {
424  QueueRecord& qr = it->second;
425  if (queue_policy_->unload_incoming(*this, it->first, gid__, qr.size))
426  {
427  log->debug("Unloading queue: {} <- {}", gid__, it->first);
428  qr.external = storage_->put(in_qrs.queues[it->first]);
429  }
430  }
431  }

434 void
435 diy::Master::
436 unload_outgoing(int gid__)
437 {
438  OutgoingQueuesRecord& out_qr = outgoing_[gid__];
439 
440  size_t out_queues_size = sizeof(size_t); // map size
441  size_t count = 0;
442  // count the size of the queues we need to pack
443  for (auto& rec : out_qr.queues)
444  {
445  if (rec.first.proc == comm_.rank()) continue;
446 
447  out_queues_size += sizeof(BlockID); // target
448  out_queues_size += Serialization<MemoryBuffer>::size(rec.second); // buffer contents
449  ++count;
450  }
451  if (queue_policy_->unload_outgoing(*this, gid__, out_queues_size - sizeof(size_t)))
452  {
453  log->debug("Unloading outgoing queues: {} -> ...; size = {}\n", gid__, out_queues_size);
454  MemoryBuffer bb; bb.reserve(out_queues_size);
455  diy::save(bb, count);
456 
457  // pack queues going to a remote proc into bb; queues going to a
458  // different block on our rank, stay separated, recorded in external_local
459  for (auto it = out_qr.queues.begin(); it != out_qr.queues.end();)
460  {
461  auto bid = it->first;
462  auto& buffer = it->second;
463  if (bid.proc == comm_.rank())
464  {
465  // treat as incoming
466  if (queue_policy_->unload_incoming(*this, gid__, bid.gid, buffer.size()))
467  {
468  QueueRecord& qr = out_qr.external_local[bid];
469  qr.size = buffer.size();
470  qr.external = storage_->put(buffer);
471 
472  out_qr.queues.erase(it++);
473  } ++it; // else keep in memory
474  } else
475  {
476  diy::save(bb, bid);
477  diy::save(bb, buffer);
478 
479  out_qr.queues.erase(it++);
480  }
481  }
482 
483  // TODO: this mechanism could be adjusted for direct saving to disk
484  // (without intermediate binary buffer serialization)
485  out_qr.external = storage_->put(bb);
486  }
487 }

668 template<class Block>
669 void
670 diy::Master::
671 iexchange_(const ICallback<Block>& f)
672 {
673  auto scoped = prof.scoped("iexchange");
674 
675  // prepare for next round
676  incoming_.erase(exchange_round_);
677  ++exchange_round_;
678 
679  IExchangeInfo iexchange(size(), comm_);
680  iexchange.add_work(size()); // start with one work unit for each block
681  comm_.barrier(); // make sure that everyone's original work is accounted for
682 
683  int global_work_ = -1;
684 
685  do
686  {
687  for (size_t i = 0; i < size(); i++) // for all blocks
688  {
689 
690  icommunicate(&iexchange); // TODO: separate comm thread std::thread t(icommunicate);
691  ProxyWithLink cp = proxy(i, &iexchange);
692 
693  prof << "callback";
694  bool done = f(block<Block>(i), cp);
695  prof >> "callback";
696 
697  int nundeq_after = 0;
698  int nunenq_after = 0;
699  for (size_t j = 0; j < cp.link()->size(); j++)
700  {
701  if (cp.incoming(cp.link()->target(j).gid))
702  ++nundeq_after;
703  if (cp.outgoing(cp.link()->target(j)).size())
704  ++nunenq_after;
705  }
706 
707  done &= (nundeq_after == 0);
708  done &= (nunenq_after == 0);
709 
710  int gid = cp.gid();
711  if (iexchange.done[gid] != done)
712  {
713  iexchange.done[gid] = done;
714  if (done)
715  {
716  int work = iexchange.dec_work();
717  log->debug("[{}] Decrementing work when switching done after callback, for {}: work = {}\n", comm_.rank(), gid, work);
718  }
719  else
720  {
721  int work = iexchange.inc_work();
722  log->debug("[{}] Incrementing work when switching done after callback, for {}: work = {}\n", comm_.rank(), gid, work);
723  }
724  }
725  }
726 
727  global_work_ = iexchange.global_work();
728  } while (global_work_ > 0);
729  log->debug("[{}] ==== Leaving iexchange ====\n", iexchange.comm.rank());
730 
731  outgoing_.clear();
732 }

747 // pseudocode for rexchange protocol based on NBX algorithm of Hoefler et al.,
748 // Scalable Communication Protocols for Dynamic Sparse Data Exchange, 2010.
749 //
750 // rcomm_exchange()
751 // {
752 // while (!done)
753 // while (sends_in_flight < limit_on_queues_in_memory and there are unprocessed queues)
754 // q = next outgoing queue (going over the in-memory queues first)
755 // if (q not in memory)
756 // load q
757 // issend(q)
758 //
759 // test all requests
760 // if (iprobe)
761 // recv
762 // if (barrier_active)
763 // if (test barrier)
764 // done = true
765 // else
766 // if (all sends finished and all queues have been processed (including out-of-core queues))
767 // ibarrier
768 // barrier_active = true
769 // }
770 //
771 void
772 diy::Master::
773 rcomm_exchange()
774 {
775  bool done = false;
776  bool ibarr_act = false;
777  mpi::request ibarr_req; // mpi request associated with ibarrier
778 
779  // make a list of outgoing queues to send (the ones in memory come first)
780  auto gid_order = order_gids();
781 
782  while (!done)
783  {
784  send_outgoing_queues(gid_order, true, 0);
785 
786  // kick requests
787  nudge();
788 
789  check_incoming_queues();
790  if (ibarr_act)
791  {
792  if (ibarr_req.test())
793  done = true;
794  }
795  else
796  {
797  if (gid_order.empty() && inflight_sends().empty())
798  {
799  ibarr_req = comm_.ibarrier();
800  ibarr_act = true;
801  }
802  }
803  } // while !done
804 }

815  {
816  OutgoingQueuesRecord& out = it->second;

819  else

831  order.limit = std::max((size_t) 1, order.size() / size() * limit_);

997 void
998 diy::Master::
999 send_different_rank(int from, int to, int proc, MemoryBuffer& bb, bool remote, IExchangeInfo* iexchange)
1000 {
1001  static const size_t MAX_MPI_MESSAGE_COUNT = INT_MAX;
1002 
1003  // sending to a different rank
1004  std::shared_ptr<MemoryBuffer> buffer = std::make_shared<MemoryBuffer>();
1005  buffer->swap(bb);
1006 
1007  MessageInfo info{from, to, exchange_round_};
1008  // size fits in one message
1009  if (Serialization<MemoryBuffer>::size(*buffer) + Serialization<MessageInfo>::size(info) <= MAX_MPI_MESSAGE_COUNT)
1010  {
1011  diy::save(*buffer, info);
1012 
1013  inflight_sends().emplace_back();
1014  auto& inflight_send = inflight_sends().back();
1015 
1016  inflight_send.info = info;
1017  if (remote || iexchange)
1018  {
1019  if (iexchange)
1020  {
1021  int work = iexchange->inc_work();
1022  log->debug("[{}] Incrementing work when sending queue: work = {}\n", comm_.rank(), work);
1023  }
1024  inflight_send.request = comm_.issend(proc, tags::queue, buffer->buffer);
1025  }
1026  else
1027  inflight_send.request = comm_.isend(proc, tags::queue, buffer->buffer);
1028  inflight_send.message = buffer;
1029  }
1030  else // large message gets broken into chunks
1031  {
1032  int npieces = static_cast<int>((buffer->size() + MAX_MPI_MESSAGE_COUNT - 1)/MAX_MPI_MESSAGE_COUNT);
1033 
1034  // first send the head
1035  std::shared_ptr<MemoryBuffer> hb = std::make_shared<MemoryBuffer>();
1036  diy::save(*hb, buffer->size());
1037  diy::save(*hb, info);
1038 
1039  inflight_sends().emplace_back();
1040  auto& inflight_send = inflight_sends().back();
1041 
1042  inflight_send.info = info;
1043  if (remote || iexchange)
1044  {
1045  // add one unit of work for the entire large message (upon sending the head, not the individual pieces below)
1046  if (iexchange)
1047  {
1048  int work = iexchange->inc_work();
1049  log->debug("[{}] Incrementing work when sending the first piece: work = {}\n", comm_.rank(), work);
1050  }
1051  inflight_send.request = comm_.issend(proc, tags::piece, hb->buffer);
1052  }
1053  else
1054  inflight_send.request = comm_.isend(proc, tags::piece, hb->buffer);
1055  inflight_send.message = hb;
1056 
1057  // send the message pieces
1058  size_t msg_buff_idx = 0;
1059  for (int i = 0; i < npieces; ++i, msg_buff_idx += MAX_MPI_MESSAGE_COUNT)
1060  {
1061  int tag = (i == (npieces - 1)) ? tags::queue : tags::piece; // last piece is marked as queue, to indicate that we are done
1062 
1063  detail::VectorWindow<char> window;
1064  window.begin = &buffer->buffer[msg_buff_idx];
1065  window.count = std::min(MAX_MPI_MESSAGE_COUNT, buffer->size() - msg_buff_idx);
1066 
1067  inflight_sends().emplace_back();
1068  auto& inflight_send = inflight_sends().back();
1069 
1070  inflight_send.info = info;
1071  if (remote || iexchange)
1072  inflight_send.request = comm_.issend(proc, tag, window);
1073  else
1074  inflight_send.request = comm_.isend(proc, tag, window);
1075  inflight_send.message = buffer;
1076  }
1077  } // large message broken into pieces
1078 }
