24  FMT_ENABLE_IF(!std::is_same<From, To>::value &&
25  std::numeric_limits<From>::is_signed ==

27 FMT_CONSTEXPR To lossless_integral_conversion(const From from, int& ec) {
28  ec = 0;
29  using F = std::numeric_limits<From>;
30  using T = std::numeric_limits<To>;
31  static_assert(F::is_integer, "From must be integral");
32  static_assert(T::is_integer, "To must be integral");
33 
34  // A and B are both signed, or both unsigned.
35  if (F::digits <= T::digits) {
36  // From fits in To without any problem.
37  } else {
38  // From does not always fit in To, resort to a dynamic check.
39  if (from < T::min() || from > T::max()) {
40  // outside range.
41  ec = 1;
42  return {};
43  }
44  }
45  return static_cast<To>(from);
46 }

57  ec = 0;
58  using F = std::numeric_limits<From>;
59  using T = std::numeric_limits<To>;
60  static_assert(F::is_integer, "From must be integral");
61  static_assert(T::is_integer, "To must be integral");
62 
63  if (F::is_signed && !T::is_signed) {
64  // From may be negative, not allowed!
65  if (from < 0) {
66  ec = 1;
67  return {};
68  }
69 
70  // From is positive. Can it always fit in To?
71  if (F::digits <= T::digits) {
72  // yes, From always fits in To.
73  } else {
74  // from may not fit in To, we have to do a dynamic check
75  if (from > static_cast<From>(T::max())) {
76  ec = 1;
77  return {};
78  }
79  }
80  }
81 
82  if (!F::is_signed && T::is_signed) {
83  // can from be held in To?
84  if (F::digits < T::digits) {
85  // yes, From always fits in To.
86  } else {
87  // from may not fit in To, we have to do a dynamic check
88  if (from > static_cast<From>(T::max())) {
89  // outside range.
90  ec = 1;
91  return {};
92  }
93  }
94  }

97  return static_cast<To>(from);

125  ec = 0;

127  static_assert(std::is_floating_point<From>::value, "From must be floating");
128  static_assert(std::is_floating_point<To>::value, "To must be floating");

137  return {};

155 template <typename To, typename FromRep, typename FromPeriod,
156  FMT_ENABLE_IF(std::is_integral<FromRep>::value),
157  FMT_ENABLE_IF(std::is_integral<typename To::rep>::value)>
158 To safe_duration_cast(std::chrono::duration<FromRep, FromPeriod> from,
159  int& ec) {
160  using From = std::chrono::duration<FromRep, FromPeriod>;
161  ec = 0;
162  // the basic idea is that we need to convert from count() in the from type
163  // to count() in the To type, by multiplying it with this:
164  using Factor = std::ratio_divide<typename From::period, typename To::period>;
165 
166  static_assert(Factor::num > 0, "num must be positive");
167  static_assert(Factor::den > 0, "den must be positive");
168 
169  // the conversion is like this: multiply from.count() with Factor::num
170  // /Factor::den and convert it to To::rep, all this without
171  // overflow/underflow. let's start by finding a suitable type that can hold
172  // both To, From and Factor::num
173  using IntermediateRep =
174  typename std::common_type<typename From::rep, typename To::rep,
175  decltype(Factor::num)>::type;
176 
177  // safe conversion to IntermediateRep
178  IntermediateRep count =
179  lossless_integral_conversion<IntermediateRep>(from.count(), ec);
180  if (ec) {
181  return {};
182  }
183  // multiply with Factor::num without overflow or underflow
184  if (Factor::num != 1) {
185  constexpr auto max1 =
186  std::numeric_limits<IntermediateRep>::max() / Factor::num;
187  if (count > max1) {
188  ec = 1;
189  return {};
190  }
191  constexpr auto min1 =
192  std::numeric_limits<IntermediateRep>::min() / Factor::num;
193  if (count < min1) {
194  ec = 1;
195  return {};
196  }
197  count *= Factor::num;
198  }
199 
200  // this can't go wrong, right? den>0 is checked earlier.
201  if (Factor::den != 1) {
202  count /= Factor::den;
203  }
204  // convert to the to type, safely
205  using ToRep = typename To::rep;
206  const ToRep tocount = lossless_integral_conversion<ToRep>(count, ec);
207  if (ec) {
208  return {};
209  }
210  return To{tocount};
211 }

216 template <typename To, typename FromRep, typename FromPeriod,
217  FMT_ENABLE_IF(std::is_floating_point<FromRep>::value),
218  FMT_ENABLE_IF(std::is_floating_point<typename To::rep>::value)>
219 To safe_duration_cast(std::chrono::duration<FromRep, FromPeriod> from,
220  int& ec) {
221  using From = std::chrono::duration<FromRep, FromPeriod>;
222  ec = 0;
223  if (std::isnan(from.count())) {
224  // nan in, gives nan out. easy.
225  return To{std::numeric_limits<typename To::rep>::quiet_NaN()};
226  }
227  // maybe we should also check if from is denormal, and decide what to do about
228  // it.
229 
230  // +-inf should be preserved.
231  if (std::isinf(from.count())) {
232  return To{from.count()};
233  }
234 
235  // the basic idea is that we need to convert from count() in the from type
236  // to count() in the To type, by multiplying it with this:
237  using Factor = std::ratio_divide<typename From::period, typename To::period>;
238 
239  static_assert(Factor::num > 0, "num must be positive");
240  static_assert(Factor::den > 0, "den must be positive");
241 
242  // the conversion is like this: multiply from.count() with Factor::num
243  // /Factor::den and convert it to To::rep, all this without
244  // overflow/underflow. let's start by finding a suitable type that can hold
245  // both To, From and Factor::num
246  using IntermediateRep =
247  typename std::common_type<typename From::rep, typename To::rep,
248  decltype(Factor::num)>::type;
249 
250  // force conversion of From::rep -> IntermediateRep to be safe,
251  // even if it will never happen be narrowing in this context.
252  IntermediateRep count =
253  safe_float_conversion<IntermediateRep>(from.count(), ec);
254  if (ec) {
255  return {};
256  }
257 
258  // multiply with Factor::num without overflow or underflow
259  if (Factor::num != 1) {
260  constexpr auto max1 = std::numeric_limits<IntermediateRep>::max() /
261  static_cast<IntermediateRep>(Factor::num);
262  if (count > max1) {
263  ec = 1;
264  return {};
265  }
266  constexpr auto min1 = std::numeric_limits<IntermediateRep>::lowest() /
267  static_cast<IntermediateRep>(Factor::num);
268  if (count < min1) {
269  ec = 1;
270  return {};
271  }
272  count *= static_cast<IntermediateRep>(Factor::num);
273  }
274 
275  // this can't go wrong, right? den>0 is checked earlier.
276  if (Factor::den != 1) {
277  using common_t = typename std::common_type<IntermediateRep, intmax_t>::type;
278  count /= static_cast<common_t>(Factor::den);
279  }
280 
281  // convert to the to type, safely
282  using ToRep = typename To::rep;
283 
284  const ToRep tocount = safe_float_conversion<ToRep>(count, ec);
285  if (ec) {
286  return {};
287  }
288  return To{tocount};
