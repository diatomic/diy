300  // Lexicographically compare this string reference to other.
301  int compare(basic_string_view other) const {
302  size_t str_size = size_ < other.size_ ? size_ : other.size_;
303  int result = std::char_traits<Char>::compare(data_, other.data_, str_size);

913 template <typename Visitor, typename Context>
914 FMT_CONSTEXPR auto visit_format_arg(Visitor&& vis,
915  const basic_format_arg<Context>& arg)
916  -> decltype(vis(0)) {
917  using char_type = typename Context::char_type;
918  switch (arg.type_) {
919  case internal::none_type:
920  break;
921  case internal::named_arg_type:
922  FMT_ASSERT(false, "invalid argument type");
923  break;
924  case internal::int_type:
925  return vis(arg.value_.int_value);
926  case internal::uint_type:
927  return vis(arg.value_.uint_value);
928  case internal::long_long_type:
929  return vis(arg.value_.long_long_value);
930  case internal::ulong_long_type:
931  return vis(arg.value_.ulong_long_value);
932  case internal::bool_type:
933  return vis(arg.value_.bool_value);
934  case internal::char_type:
935  return vis(arg.value_.char_value);
936  case internal::double_type:
937  return vis(arg.value_.double_value);
938  case internal::long_double_type:
939  return vis(arg.value_.long_double_value);
940  case internal::cstring_type:
941  return vis(arg.value_.string.data);
942  case internal::string_type:
943  return vis(basic_string_view<char_type>(arg.value_.string.data,
944  arg.value_.string.size));
945  case internal::pointer_type:
946  return vis(arg.value_.pointer);
947  case internal::custom_type:
948  return vis(typename basic_format_arg<Context>::handle(arg.value_.custom));
949  }
950  return vis(monostate());
951 }

1130 template <typename Context> class basic_format_args {
1131  public:
1132  using size_type = int;
1133  using format_arg = basic_format_arg<Context>;
1134 
1135  private:
1136  // To reduce compiled code size per formatting function call, types of first
1137  // max_packed_args arguments are passed in the types_ field.
1138  unsigned long long types_;
1139  union {
1140  // If the number of arguments is less than max_packed_args, the argument
1141  // values are stored in values_, otherwise they are stored in args_.
1142  // This is done to reduce compiled code size as storing larger objects
1143  // may require more code (at least on x86-64) even if the same amount of
1144  // data is actually copied to stack. It saves ~10% on the bloat test.
1145  const internal::value<Context>* values_;
1146  const format_arg* args_;
1147  };
1148 
1149  bool is_packed() const { return (types_ & internal::is_unpacked_bit) == 0; }
1150 
1151  internal::type type(int index) const {
1152  int shift = index * 4;
1153  return static_cast<internal::type>((types_ & (0xfull << shift)) >> shift);
1154  }
1155 
1156  friend class internal::arg_map<Context>;
1157 
1158  void set_data(const internal::value<Context>* values) { values_ = values; }
1159  void set_data(const format_arg* args) { args_ = args; }
1160 
1161  format_arg do_get(int index) const {
1162  format_arg arg;
1163  if (!is_packed()) {
1164  auto num_args = max_size();
1165  if (index < num_args) arg = args_[index];
1166  return arg;
1167  }
1168  if (index > internal::max_packed_args) return arg;
1169  arg.type_ = type(index);
1170  if (arg.type_ == internal::none_type) return arg;
1171  internal::value<Context>& val = arg.value_;
1172  val = values_[index];
1173  return arg;
1174  }
1175 
1176  public:
1177  basic_format_args() : types_(0) {}
1178 
1184  template <typename... Args>
1185  basic_format_args(const format_arg_store<Context, Args...>& store)
1186  : types_(static_cast<unsigned long long>(store.types)) {
1187  set_data(store.data_);
1188  }
1189 
1195  basic_format_args(const format_arg* args, int count)
1196  : types_(internal::is_unpacked_bit | internal::to_unsigned(count)) {
1197  set_data(args);
1198  }
1199 
1201  format_arg get(int index) const {
1202  format_arg arg = do_get(index);
1203  if (arg.type_ == internal::named_arg_type)
1204  arg = arg.value_.named_arg->template deserialize<Context>();
1205  return arg;
1206  }
1207

1209  unsigned long long max_packed = internal::max_packed_args;
1210  return static_cast<int>(is_packed() ? max_packed
1211  : types_ & ~internal::is_unpacked_bit);
1212  }
