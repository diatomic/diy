79  header = header.substr(11, nl - 11 + 1);

82  int loc1, loc2;
83 
84  //fortran order

88  //shape

93  if(str_shape[str_shape.size()-1] == ',') ndims = 1;

99  str_shape = str_shape.substr(loc1+1);
100  }
