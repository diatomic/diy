126 template <typename Char> struct formatter<std::tm, Char> {
127  template <typename ParseContext>
128  auto parse(ParseContext& ctx) -> decltype(ctx.begin()) {
129  auto it = ctx.begin();
130  if (it != ctx.end() && *it == ':') ++it;
131  auto end = it;
132  while (end != ctx.end() && *end != '}') ++end;
133  tm_format.reserve(internal::to_unsigned(end - it + 1));
134  tm_format.append(it, end);
135  tm_format.push_back('\0');
136  return end;
137  }

139  template <typename FormatContext>
140  auto format(const std::tm& tm, FormatContext& ctx) -> decltype(ctx.out()) {
141  basic_memory_buffer<Char> buf;
142  std::size_t start = buf.size();
143  for (;;) {
144  std::size_t size = buf.capacity() - start;
145  std::size_t count =
146  internal::strftime(&buf[start], size, &tm_format[0], &tm);
147  if (count != 0) {
148  buf.resize(start + count);
149  break;
150  }
151  if (size >= tm_format.size() * 256) {
152  // If the buffer is 256 times larger than the format string, assume
153  // that `strftime` gives an empty result. There doesn't seem to be a
154  // better way to distinguish the two cases:
155  // https://github.com/fmtlib/fmt/issues/367
156  break;
157  }
158  const std::size_t MIN_GROWTH = 10;
159  buf.reserve(buf.capacity() + (size > MIN_GROWTH ? size : MIN_GROWTH));
160  }
161  return std::copy(buf.begin(), buf.end(), ctx.out());
162  }
163 
164  basic_memory_buffer<Char> tm_format;
165 };

201 // Parses a put_time-like format string and invokes handler actions.
202 template <typename Char, typename Handler>
203 FMT_CONSTEXPR const Char* parse_chrono_format(const Char* begin,
204  const Char* end,
205  Handler&& handler) {
206  auto ptr = begin;
207  while (ptr != end) {
208  auto c = *ptr;
209  if (c == '}') break;
210  if (c != '%') {
211  ++ptr;
212  continue;
213  }
214  if (begin != ptr) handler.on_text(begin, ptr);
215  ++ptr; // consume '%'
216  if (ptr == end) FMT_THROW(format_error("invalid format"));
217  c = *ptr++;
218  switch (c) {
219  case '%':
220  handler.on_text(ptr - 1, ptr);
221  break;
222  case 'n': {
223  const char newline[] = "\n";
224  handler.on_text(newline, newline + 1);
225  break;
226  }
227  case 't': {
228  const char tab[] = "\t";
229  handler.on_text(tab, tab + 1);
230  break;
231  }
232  // Day of the week:
233  case 'a':
234  handler.on_abbr_weekday();
235  break;
236  case 'A':
237  handler.on_full_weekday();
238  break;
239  case 'w':
240  handler.on_dec0_weekday(numeric_system::standard);
241  break;
242  case 'u':
243  handler.on_dec1_weekday(numeric_system::standard);
244  break;
245  // Month:
246  case 'b':
247  handler.on_abbr_month();
248  break;
249  case 'B':
250  handler.on_full_month();
251  break;
252  // Hour, minute, second:
253  case 'H':
254  handler.on_24_hour(numeric_system::standard);
255  break;
256  case 'I':
257  handler.on_12_hour(numeric_system::standard);
258  break;
259  case 'M':
260  handler.on_minute(numeric_system::standard);
261  break;
262  case 'S':
263  handler.on_second(numeric_system::standard);
264  break;
265  // Other:
266  case 'c':
267  handler.on_datetime(numeric_system::standard);
268  break;
269  case 'x':
270  handler.on_loc_date(numeric_system::standard);
271  break;
272  case 'X':
273  handler.on_loc_time(numeric_system::standard);
274  break;
275  case 'D':
276  handler.on_us_date();
277  break;
278  case 'F':
279  handler.on_iso_date();
280  break;
281  case 'r':
282  handler.on_12_hour_time();
283  break;
284  case 'R':
285  handler.on_24_hour_time();
286  break;
287  case 'T':
288  handler.on_iso_time();
289  break;
290  case 'p':
291  handler.on_am_pm();
292  break;
293  case 'Q':
294  handler.on_duration_value();
295  break;
296  case 'q':
297  handler.on_duration_unit();
298  break;
299  case 'z':
300  handler.on_utc_offset();
301  break;
302  case 'Z':
303  handler.on_tz_name();
304  break;
305  // Alternative representation:
306  case 'E': {
307  if (ptr == end) FMT_THROW(format_error("invalid format"));
308  c = *ptr++;
309  switch (c) {
310  case 'c':
311  handler.on_datetime(numeric_system::alternative);
312  break;
313  case 'x':
314  handler.on_loc_date(numeric_system::alternative);
315  break;
316  case 'X':
317  handler.on_loc_time(numeric_system::alternative);
318  break;
319  default:
320  FMT_THROW(format_error("invalid format"));
321  }
322  break;
323  }
324  case 'O':
325  if (ptr == end) FMT_THROW(format_error("invalid format"));
326  c = *ptr++;
327  switch (c) {
328  case 'w':
329  handler.on_dec0_weekday(numeric_system::alternative);
330  break;
331  case 'u':
332  handler.on_dec1_weekday(numeric_system::alternative);
333  break;
334  case 'H':
335  handler.on_24_hour(numeric_system::alternative);
336  break;
337  case 'I':
338  handler.on_12_hour(numeric_system::alternative);
339  break;
340  case 'M':
341  handler.on_minute(numeric_system::alternative);
342  break;
343  case 'S':
344  handler.on_second(numeric_system::alternative);
345  break;
346  default:
347  FMT_THROW(format_error("invalid format"));
348  }
349  break;
350  default:
351  FMT_THROW(format_error("invalid format"));
352  }
353  begin = ptr;
354  }
355  if (begin != ptr) handler.on_text(begin, ptr);
356  return ptr;
357 }

453 template <typename Rep, typename Period,
454  FMT_ENABLE_IF(std::is_integral<Rep>::value)>
455 inline std::chrono::duration<Rep, std::milli> get_milliseconds(
456  std::chrono::duration<Rep, Period> d) {
457  // this may overflow and/or the result may not fit in the
458  // target type.
459 #if FMT_SAFE_DURATION_CAST
460  using CommonSecondsType =
461  typename std::common_type<decltype(d), std::chrono::seconds>::type;
462  const auto d_as_common = fmt_safe_duration_cast<CommonSecondsType>(d);
463  const auto d_as_whole_seconds =
464  fmt_safe_duration_cast<std::chrono::seconds>(d_as_common);
465  // this conversion should be nonproblematic
466  const auto diff = d_as_common - d_as_whole_seconds;
467  const auto ms =
468  fmt_safe_duration_cast<std::chrono::duration<Rep, std::milli>>(diff);
469  return ms;
470 #else
471  auto s = std::chrono::duration_cast<std::chrono::seconds>(d);
472  return std::chrono::duration_cast<std::chrono::milliseconds>(d - s);
473 #endif
474 }

772  auto begin = ctx.begin(), end = ctx.end();
773  if (begin == end || *begin == '}') return {begin, begin};
774  spec_handler handler{*this, ctx, format_str};
775  begin = internal::parse_align(begin, end, handler);
776  if (begin == end) return {begin, begin};
777  begin = internal::parse_width(begin, end, handler);

800  template <typename FormatContext>
801  auto format(const duration& d, FormatContext& ctx) -> decltype(ctx.out()) {
802  auto begin = format_str.begin(), end = format_str.end();
803  // As a possible future optimization, we could avoid extra copying if width
804  // is not specified.
805  basic_memory_buffer<Char> buf;
806  auto out = std::back_inserter(buf);
807  using range = internal::output_range<decltype(ctx.out()), Char>;
808  internal::basic_writer<range> w(range(ctx.out()));
809  internal::handle_dynamic_spec<internal::width_checker>(
810  specs.width, width_ref, ctx, format_str.begin());
811  internal::handle_dynamic_spec<internal::precision_checker>(
812  precision, precision_ref, ctx, format_str.begin());
813  if (begin == end || *begin == '}') {
814  out = internal::format_chrono_duration_value(out, d.count(), precision);
815  internal::format_chrono_duration_unit<Period>(out);
816  } else {
817  internal::chrono_formatter<FormatContext, decltype(out), Rep, Period> f(
818  ctx, out, d);
819  f.precision = precision;
820  parse_chrono_format(begin, end, f);
821  }
822  w.write(buf.data(), buf.size(), specs);
823  return w.out();
824  }
