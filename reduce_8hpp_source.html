108 void reduce(Master& master,
109  const Assigner& assigner,
110  const Partners& partners,
111  const Reduce& reduce,
112  const Skip& skip)
113 {
114  int original_expected = master.expected();
115 
116  unsigned round;
117  for (round = 0; round < partners.rounds(); ++round)
118  {
119  //fprintf(stderr, "== Round %d\n", round);
120  master.foreach(detail::ReductionFunctor<Reduce,Partners>(round, reduce, partners, assigner),
121  detail::SkipInactiveOr<Partners,Skip>(round, partners, skip));
122  master.execute();
123 
124  int expected = 0;
125  for (int i = 0; i < master.size(); ++i)
126  {
127  if (partners.active(round + 1, master.gid(i), master))
128  {
129  std::vector<int> incoming_gids;
130  partners.incoming(round + 1, master.gid(i), incoming_gids, master);
131  expected += incoming_gids.size();
132  master.incoming(master.gid(i)).clear();
133  }
134  }
135  master.set_expected(expected);
136  master.flush();
137  }
138  // final round
139  //fprintf(stderr, "== Round %d\n", round);
140  master.foreach(detail::ReductionFunctor<Reduce,Partners>(round, reduce, partners, assigner),
141  detail::SkipInactiveOr<Partners,Skip>(round, partners, skip));
142 
143  master.set_expected(original_expected);
144 }
